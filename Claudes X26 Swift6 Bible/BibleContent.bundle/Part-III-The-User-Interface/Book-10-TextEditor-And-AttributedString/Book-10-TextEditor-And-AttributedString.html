  1. Chapter 10: TextEditor and AttributedString

  2. TextEditor

  3. TextEditor is SwiftUI's multi-line text input. Use it when TextField with axis: .vertical is not enough.

  4. @State private var notes = ""

TextEditor(text: $notes)
    .frame(height: 200)

  5. Styling TextEditor

  6. TextEditor has a default background and some built-in padding. You will almost always want to customize it.

  7. TextEditor(text: $notes)
    .font(.body)
    .foregroundStyle(.primary)
    .scrollContentBackground(.hidden) // remove default background
    .background(.ultraThinMaterial)
    .clipShape(.rect(cornerRadius: 12))
    .frame(minHeight: 100, maxHeight: 300)

  8. Watch out: Before .scrollContentBackground(.hidden) existed, people hacked around the background with UITextView.appearance(). Do not do that. Use .scrollContentBackground(.hidden) and then apply your own .background().

  9. Placeholder Text

  10. TextEditor has no built-in placeholder. Overlay one yourself:

  11. ZStack(alignment: .topLeading) {
    if notes.isEmpty {
        Text("Write your notes here...")
            .foregroundStyle(.tertiary)
            .padding(.top, 8)
            .padding(.leading, 5)
            .allowsHitTesting(false)
    }
    TextEditor(text: $notes)
        .scrollContentBackground(.hidden)
}

  12. Line Limit

  13. TextEditor(text: $notes)
    .lineLimit(5...10) // constrain vertical growth

  14. Disabling Editing

  15. TextEditor(text: .constant(readOnlyText))
    // or
TextEditor(text: $notes)
    .disabled(true)

  16. Find and Replace

  17. Enable the system find bar (Cmd+F on Mac, find UI on iPad):

  18. @State private var isSearchPresented = false

TextEditor(text: $notes)
    .findNavigator(isPresented: $isSearchPresented)

  19. Toggle isSearchPresented from a button or toolbar item to show/hide find and replace:

  20. .toolbar {
    Button("Find", systemImage: "magnifyingglass") {
        isSearchPresented.toggle()
    }
}

  21. Watch out: .findNavigator works on TextEditor, List, and Table. It does not work on plain Text views.

  22. Tracking Changes

  23. Use onChange to react to text edits:

  24. TextEditor(text: $notes)
    .onChange(of: notes) { oldValue, newValue in
        wordCount = newValue.split(separator: " ").count
        hasUnsavedChanges = true
    }

  25. The two-parameter onChange closure (old, new) is the current API. The single-parameter version is deprecated.

  27. Text Selection

  28. Control whether text is selectable:

  29. Text("Read-only but selectable")
    .textSelection(.enabled)

  30. Apply to a container to enable selection for all text inside:

  31. VStack {
    Text("First line")
    Text("Second line")
    Text("Third line")
}
.textSelection(.enabled)

  33. AttributedString

  34. AttributedString is Swift's native attributed string type. It replaces most uses of NSAttributedString in SwiftUI.

  35. Creating an AttributedString

  36. var str = AttributedString("Hello, world")
str.font = .title
str.foregroundColor = .blue
str.underlineStyle = .single

Text(str)

  37. Styling Ranges

  38. var str = AttributedString("Hello, bold world")
if let range = str.range(of: "bold") {
    str[range].font = .body.bold()
    str[range].foregroundColor = .red
}
Text(str)

  39. Combining AttributedStrings

  40. var greeting = AttributedString("Hello ")
greeting.font = .headline

var name = AttributedString("Michael")
name.font = .headline
name.foregroundColor = .blue

Text(greeting + name)

  41. Available Attributes

  42. Common attributes you can set on AttributedString or its ranges:

  43. | Attribute | Type | Example | |---|---|---| | .font | Font | .body, .title | | .foregroundColor | Color | .red, .blue | | .backgroundColor | Color | .yellow | | .strikethroughStyle | Text.LineStyle | .single, .double | | .underlineStyle | Text.LineStyle | .single | | .underlineColor | Color | .red | | .kern | CGFloat | 2.0 | | .tracking | CGFloat | 1.5 | | .baselineOffset | CGFloat | 5.0 | | .link | URL | URL(string: "...") |

  45. Markdown Parsing with AttributedString

  46. AttributedString can parse Markdown directly:

  47. let markdown = "This is **bold**, this is *italic*, and this is a [link](https://example.com)"

if let attributed = try? AttributedString(markdown: markdown) {
    Text(attributed)
}

  48. Parsing Options

  49. let options = AttributedString.MarkdownParsingOptions(
    interpretedSyntax: .inlineOnlyPreservingWhitespace
)

let str = try? AttributedString(markdown: source, options: options)

  50. Interpreted syntax options:

    • .inlineOnlyPreservingWhitespace -- inline markdown only, keeps whitespace as-is (best for single-line UI text)
    • .full -- full CommonMark parsing including block elements

  52. Handling Parse Failures

  53. do {
    let attributed = try AttributedString(markdown: rawText)
    Text(attributed)
} catch {
    Text(rawText) // fall back to plain text
}

  54. Watch out: The markdown parser is strict. Malformed markdown throws an error rather than rendering partially. Always have a fallback.

  56. Custom Attributes

  57. You can define your own attributes for domain-specific styling.

  58. Define the Attribute

  59. enum HighlightAttribute: CodableAttributedStringKey {
    typealias Value = Bool
    static let name = "highlight"
}

extension AttributeScopes {
    struct AppAttributes: AttributeScope {
        let highlight: HighlightAttribute
    }

    var app: AppAttributes.Type { AppAttributes.self }
}

  60. Use the Custom Attribute

  61. var str = AttributedString("Important note")
str.highlight = true

  62. Render with Custom Styling

  63. Custom attributes do not auto-style in Text. You walk the runs and apply SwiftUI attributes based on your custom ones:

  64. func styled(_ source: AttributedString) -> AttributedString {
    var result = source
    for run in result.runs {
        if run.highlight == true {
            result[run.range].backgroundColor = .yellow
            result[run.range].font = .body.bold()
        }
    }
    return result
}

  65. Walking Runs

  66. AttributedString.runs gives you each contiguous range of uniform attributes:

  67. for run in attributed.runs {
    print("Text: \(attributed[run.range].characters)")
    print("Font: \(run.font ?? .body)")
    print("---")
}

  69. NSAttributedString Bridging

  70. When you need to work with UIKit APIs or older code:

  71. AttributedString to NSAttributedString

  72. let modern = AttributedString("Hello")
let legacy = NSAttributedString(modern)

  73. NSAttributedString to AttributedString

  74. let legacy = NSAttributedString(string: "Hello", attributes: [
    .foregroundColor: UIColor.red,
    .font: UIFont.boldSystemFont(ofSize: 18)
])

let modern = AttributedString(legacy)
Text(modern)

  75. Watch out: Not all NSAttributedString attributes have AttributedString equivalents. UIKit-specific attributes like .paragraphStyle may not translate cleanly. Test the conversion.

  76. When You Still Need NSAttributedString

    • Core Text rendering
    • UIKit text views wrapped in UIViewRepresentable
    • Specific paragraph style control (line height, alignment, tab stops) not yet in AttributedString

  79. Practical Patterns

  80. Rich Text Display

  81. struct RichTextView: View {
    let markdown: String

    var body: some View {
        if let attributed = try? AttributedString(markdown: markdown) {
            Text(attributed)
                .font(.body)
                .textSelection(.enabled)
        } else {
            Text(markdown)
                .font(.body)
        }
    }
}

  82. Notes Editor with Word Count

  83. struct NotesEditor: View {
    @Binding var text: String
    @State private var wordCount = 0
    @State private var showFind = false
    @FocusState private var isFocused: Bool

    var body: some View {
        VStack(alignment: .leading, spacing: 8) {
            TextEditor(text: $text)
                .focused($isFocused)
                .font(.body)
                .scrollContentBackground(.hidden)
                .background(.background.secondary)
                .clipShape(.rect(cornerRadius: 8))
                .findNavigator(isPresented: $showFind)
                .onChange(of: text) { _, newValue in
                    wordCount = newValue.split(separator: " ").count
                }

            HStack {
                Text("\(wordCount) words")
                    .font(.caption)
                    .foregroundStyle(.secondary)

                Spacer()

                Button("Find") { showFind.toggle() }
                    .font(.caption)
            }
        }
    }
}

  84. Highlightable Log Viewer

  85. func highlightErrors(in log: String) -> AttributedString {
    var result = AttributedString(log)

    let lines = log.components(separatedBy: "\n")
    var position = result.startIndex

    for line in lines {
        let lineEnd = result.index(position, offsetByCharacters: line.count)
        let range = position..<lineEnd

        if line.contains("ERROR") {
            result[range].foregroundColor = .red
            result[range].font = .body.bold()
        } else if line.contains("WARNING") {
            result[range].foregroundColor = .orange
        } else {
            result[range].foregroundColor = .secondary
        }

        // move past the newline
        if lineEnd < result.endIndex {
            position = result.index(lineEnd, offsetByCharacters: 1)
        } else {
            break
        }
    }

    return result
}
